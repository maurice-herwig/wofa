# Die Idee ist, da u1 und u3 beliebige Wörter des Alphabets. Diese sind durch U1 und U3 beliebig wählbar.
# Weiterhin ist als übergang zwischen u1 und u2 das "a" benötigt, sowie zwischen u2 und u3 das "b" benötigt ist.
# Da wir in L3 sind ist gefordert das die Länge des Wortes U2 gleich n=3 ist, was durch die Zustände q_1 bis q_4 erfolgt.
# In q_4 wird dann nochmals geprüft ob es nun u1au2b ist, wenn nicht wird alles als ein verlängertes u1 gedeutet.
# Dadurch wird garantiert das die Übergänge mit a und b jeweils vor und nach einem 3 langem Wort u2 erfolgen.
# Dies bedeutet wir haben u1a...bu3 wobei die drei Punkte für die Zeichen des Wortes u2 stehen.

input_alphabet = a, b

start_states = q_0

transitions = 	U1, a -> U1; q_1
		U1, b -> U1

		q_1, a -> q_2
		q_1, b -> q_2

		q_2, a -> q_3
		q_2, b -> q_3

		q_3, a -> q_4
		q_3, b -> q_4
		
		q_4, b -> U3
		q_4, a -> U1

		U3, a -> U3
		U3, b -> U3

acc_states = U3
