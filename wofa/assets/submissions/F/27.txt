input_alphabet = a, b

start_states = q_0

transitions = q_0, a  -> q_1
              q_0, b  -> q_0

              q_1, a  -> q_2
              q_1, b  -> q_2

              q_2, a  -> q_3
              q_2, b  -> q_3

              q_3, a  -> q_4
              q_3, b  -> q_4

              q_4, a  -> q_1
              q_4, b  -> q_5

              q_5, a  -> q_5
              q_5, b  -> q_5

acc_states = q_5

# Begründung:
#
# Die Wörter der Sprache Ln haben mindestens n+2 zeichen. Also für n=3 hat L3 5 Zeichen (a u2 b) falls u1 und u2 das leere Wort sind.
# Der DFA erkennt die Sprache, da er im Zustand q_1 jede Eingabe als mögliches u2 erkennt.
# Die Eingaben von q_1 -> q_2 -> q_3 -> q_4 sind egal da sie nur dazu dienen die Länge von u2 fest zulegen und nicht den Inhalt.
# In q_4 entscheidet sich ob das mögliche u2 tatsächlich ein u2 ist. Für eine Eingabe von b in q_4 geht der DFA in den Endzustand q_5 und alle weiteren Zeichen gehören zu u3.
# Für die Eingabe von a n q_4 springt der DFA zurück in q_1 und die bisherige Eingabe bis auf das letzte a wird zu u1 hinzugefügt.
